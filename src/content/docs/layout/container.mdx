A utility that sets the width of an element at the current breakpoint.

<Utility
  data={[
    {
      className: "cnt",
      properties: [
        "width: 100%;",
        "max-width: 640px;",
        "max-width: 768px;",
        "max-width: 1024px;",
        "max-width: 1280px;",
        "max-width: 1536px;",
      ],
    },
  ]}
/>
