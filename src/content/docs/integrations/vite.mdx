export const onCSS = `/* Standard version */
import 'yummacss/dist/yumma.css';`;

<Steps>
    1. Install Yumma CSS in your Vite application:

        ```bash title="Terminal"
        npm install yummacss@latest
        ```

    2. Import Yumma CSS in your main JavaScript or CSS file:

        <Code code={onCSS} lang="javascript" title="main.js" />

    3. Start your Vite development server:

        ```bash title="Terminal"
        npm run dev
        ```

    4. Begin using Yumma CSS in your Vite project:

        ```html title="index.html" {2-4}
        <body>
            <div id="app" class="ai-c bg-blue d-f jc-c h-1/1">
                <h1 class="fs-lg fw-700 tc-white">Yumma CSS + Vite</h1>
            </div>

            <script type="module" src="/src/main.js"></script>
        </body>
        ```

        ```javascript title="src/main.js"
        import 'yummacss/dist/yumma.css';

        const app = document.getElementById('app');
        ```

</Steps>
