export const onCSS = `/* Standard version */
import 'yummacss/dist/yumma.css';`;

<Steps>
    1. Install Yumma CSS in your React application:

        ```bash title="Terminal"
        npm install yummacss@latest
        ```

    2. Import Yumma CSS in your main CSS file or JS file:

        <Code code={onCSS} lang="javascript" title="index.js" />

    3. Start your React development server:

        ```bash title="Terminal"
        npm start
        ```

    4. Begin using Yumma CSS in your React project:

        ```jsx title="App.js" {6-8}
        import React from 'react';
        import 'yummacss/dist/yumma.css';

        function App() {
          return (
            <div className="ai-c bg-blue d-f jc-c h-1/1">
              <h1 className="fs-lg fw-700 tc-white">Yumma CSS + React</h1>
            </div>
          );
        }

        export default App;
        ```
</Steps>
