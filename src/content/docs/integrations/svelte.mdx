export const onCSS = `/* Standard version */
import 'yummacss/dist/yumma.css';`;

<Steps>
    1. Install Yumma CSS in your Svelte application:

        ```bash title="Terminal"
        npm install yummacss@latest
        ```

    2. Import Yumma CSS in your main CSS or JS file:

        <Code code={onCSS} lang="javascript" title="main.js" />

    3. Start your Svelte development server:

        ```bash title="Terminal"
        npm run dev
        ```

    4. Begin using Yumma CSS in your Svelte project:

        ```svelte title="App.svelte" {5-7}
        <script>
          import 'yummacss/dist/yumma.css';
        </script>

        <div class="ai-c bg-blue d-f jc-c h-1/1">
          <h1 class="fs-lg fw-700 tc-white">Yumma CSS + Svelte</h1>
        </div>
        ```

</Steps>
