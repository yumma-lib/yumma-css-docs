import { Code } from "@astrojs/starlight/components";
import { Steps } from "@astrojs/starlight/components";

export const onHTML = `<!-- Standard version -->
<link rel="stylesheet" href="styles/globals.css">`;
export const onCSS = `/* Standard version */
@import "yummacss/dist/yumma.css";`;

<Steps>
    1. Install Yumma CSS in your application:

        ```bash title="Terminal"
        npm install yummacss@latest
        ```

    2. Add `yummacss` in your main CSS file:

        <Code code={onCSS} lang="css" title="globals.css" />

    3. Include the compiled CSS file in your `<head>` element:

        <Code code={onHTML} lang="html" title="index.html" />

    4. Start your development server:

        ```bash title="Terminal"
        npm run dev
        ```
    5. Begin using Yumma CSS in your Next.js project:

        ```tsx {3-5}
        export default function Home() {
          return (
            <div class="ai-c bg-blue d-f jc-c h-1/1">
              <h1 class="fs-lg fw-700 tc-white">Yumma CSS + Next.js</h1>
            </div>
          )
        }
        ```

</Steps>
