import { Code, Steps } from "@astrojs/starlight/components";

export const onCSS = `/* Standard version */
@import "yummacss/dist/yumma.css";`;

<Steps>
    1. Install Yumma CSS in your Astro project:

        ```bash title="Terminal"
        npm install yummacss@latest
        ```

    2. Add Yumma CSS to your global styles:

        <Code code={onCSS} lang="css" title="global.css" />

    3. Ensure that your global CSS is included:

        ```html
        <head>
          <link rel="stylesheet" href="/src/styles/global.css">
        </head>
        ```

    4. Start your Astro development server:

        ```bash title="Terminal"
        npm run dev
        ```

    5. Begin using Yumma CSS in your Astro project:

        ```html title="src/pages/index.astro"
        <div class="ai-c bg-blue d-f jc-c h-1/1">
          <h1 class="fs-lg fw-700 tc-white">Yumma CSS + Astro</h1>
        </div>
        ```

</Steps>
