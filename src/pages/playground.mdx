"/Header.js": {
code: `import React, { useState } from "react";

import { IoMenu, IoClose } from "react-icons/io5";

const Header = ({ appName }) => {
  const [showDropdown, setShowDropdown] = useState(false);

  const toggleDropdown = () => {
    setShowDropdown(!showDropdown);
  };

  return (
   <header className="bg-transparent d-f fw-w py-1 sm:fw-nw sm:jc-s sm:py-0 w-full zi-50">
      <nav className="lg:px-3 mx-auto pos-r px-1 sm:ai-c sm:d-f sm:jc-b sm:px-2 w-full">
         <div className="ai-c d-f jc-b">
            <a className="f-none fs-lg fw-600 t-d-lead-1" href="/">
               {appName}
            </a>
            <div className="ml-auto sm:d-none">
               <button className="c-p ai-c bg-transparent cg-2 d-f dim-9 fw-600 jc-c rad-1" onClick={toggleDropdown}>
                  {showDropdown ?
                  <IoClose className="flex-shrink-0 size-4" /> :
                  <IoMenu className="flex-shrink-0 size-4" />}
               </button>
            </div>
         </div>
         {showDropdown && (
         <div className="fb-full fg-1 ovf-h sm:d-b">
            <div className="cg-0 d-f fd-c mt-2 rg-4 sm:ai-c sm:cg-3 sm:fd-r sm:jc-e sm:mt-0 sm:rg-0">
               <a className="fs-sm fw-600 h:t-indigo sm:py-2 t-d-lead-1" href="/">
                  Docs
               </a>
               <a className="fs-sm fw-600 h:t-indigo sm:py-2 t-d-lead-1" href="/">
                  Playground
               </a>
               <a className="fs-sm fw-600 h:t-indigo sm:py-2 t-d-lead-1" href="/">
                  Yumma CSS
               </a>
            </div>
         </div>
         )}
      </nav>
   </header>
  );
};

export default Header;
`,
