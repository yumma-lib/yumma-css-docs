# Display

Specifies the rendering behavior of an element.

<ClassList
  classData={[
    {
      className: 'd-b',
      properties: [
        'display: block;'
      ]
    },
    {
      className: 'd-i',
      properties: [
        'display: inline;'
      ]
    },
    {
      className: 'd-ib',
      properties: [
        'display: inline-block;'
      ]
    },
    {
      className: 'd-f',
      properties: [
        'display: flex;'
      ]
    },
    {
      className: 'd-if',
      properties: [
        'display: inline-flex;'
      ]
    },
    {
      className: 'd-g',
      properties: [
        'display: grid;'
      ]
    },
    {
      className: 'd-fr',
      properties: [
        'display: flow-root;'
      ]
    },
    {
      className: 'd-none',
      properties: [
        'display: none;'
      ]
    },
  ]}
/>
