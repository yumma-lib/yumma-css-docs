# Justify Content

Aligns flex or grid items horizontally.

<ClassList
  classData={[
    {
      className: 'jc-c',
      properties: [
        'justify-content: center;'
      ]
    },
    {
      className: 'jc-sa',
      properties: [
        'justify-content: space-around;'
      ]
    },
    {
      className: 'jc-sb',
      properties: [
        'justify-content: space-between;'
      ]
    },
    {
      className: 'jc-se',
      properties: [
        'justify-content: space-evenly;'
      ]
    },
    {
      className: 'jc-s',
      properties: [
        'justify-content: start;'
      ]
    },
  ]}
/>
